def solution(n, times):
    answer = float('inf')  # --1
    left = min(times) # --2
    right = max(times) * n # --3

    def get_people(maxTime, timeArr):
        maxPeople = 0
        for time in timeArr:
            maxPeople += maxTime // time
        return maxPeople

    while left <= right: # --5
        mid = (left + right) // 2
        totalPeople = get_people(mid, times)
        if totalPeople >= n: # --4
            answer = min([answer, mid])
            right = mid - 1
        else:
            left = mid + 1

    return answer
